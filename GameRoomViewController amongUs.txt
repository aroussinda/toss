import UIKit

class GameRoomViewController: UIViewController,UITableViewDelegate,UITableViewDataSource {
    
    

    
    
    //var
    
    var userName1:String?
    let avatarName = ["Hu5tl3r", "MC_Killer", "CR7", "Not_Impostor", "BabyShark", "AKINFENWA","DR'Who"]
    let avatarColor = ["Brown", "Light-blue", "Orange", "Purple", "Red", "Rose","Yellow"]
    
    //iboutlets
    
    @IBOutlet weak var helloUserLabel: UILabel!
    
    
    
    
    //ibactions
    
    
    
    
    //functions
    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        return avatarName.count
    }
    
    func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "mCell")

                let contentView = cell?.contentView
                
                let namelabel = contentView?.viewWithTag(2) as! UILabel
                let colorlabel = contentView?.viewWithTag(1) as! UILabel

                let imageView = contentView?.viewWithTag(3) as! UIImageView
                
                namelabel.text = avatarName[indexPath.row]
                colorlabel.text = avatarColor[indexPath.row]
                imageView.image = UIImage(named: avatarColor[indexPath.row])
                
                return cell!

    }
    
    //cellule onClickListener
    func tableView(_ tableView: UITableView, didSelectRowAt indexPath: IndexPath) {
    
        performSegue(withIdentifier: "secondSegue", sender: indexPath)
        
        
    }
    
    override func prepare(for segue: UIStoryboardSegue, sender: Any?) {
        
        if segue.identifier == "secondSegue" {
            let indexPath = sender as! IndexPath
            let destination = segue.destination as! DetailsViewController
            destination.PlayerName1 = avatarName[indexPath.row]
            destination.avatarColor1 = avatarColor[indexPath.row]
        }
        
    }
    
    override func viewDidLoad() {
        super.viewDidLoad()
        helloUserLabel.text = "hello \(userName1!)"
